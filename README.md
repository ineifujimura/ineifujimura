client.on('ready', () => {
  console.log(`Logged in as ${client.user.tag}!`);

  if (!interaction.isCommand()) return;

  if (interaction.commandName === 'contact') {
    await interaction.reply('You will see me on discord as Inei Fujimura#2709');
  }
});
